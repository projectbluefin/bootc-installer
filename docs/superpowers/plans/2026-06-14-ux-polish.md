```python
def get_hostname_override(self):
    """Return the hostname the user typed on the confirm screen, or None.

    Returns None if the confirm screen was never shown a hostname field
    (e.g. the disk step is hidden or no hostname was in finals).
    """
    if self._hostname_entry_row is None:

    text = self._hostname_entry_row.get_text().strip()
    return text if text else None
```

```python
_RE_FRIENDLY_LAYER = re.compile(r"Pulling image: layer (\d+)/(\d+)")

def _friendly_substep(msg: str) -> str:
    """Convert a raw fisherman substep message to human-friendly text."""
    m = _RE_FRIENDLY_LAYER.match(msg)
    if m:
        done, total = m.group(1), m.group(2)
        return _("Downloading… (%s of %s parts)") % (done, total)
    if msg.startswith("Pulling image"):
        return _("Downloading Bluefin…")
```
